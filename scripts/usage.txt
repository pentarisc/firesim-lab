# Install dependencies (once, inside Docker image or venv)
pip install -r scripts/requirements.txt

# Discover what's available
init-target --list

# Typical invocations
init-target --name my-dsp --bridge uart,fased \
    --feature verilog-blackbox --blackbox-name MyDSP

# Repeated flags style (equivalent)
init-target --name my-dsp \
            --bridge uart --bridge fased \
            --feature verilog-blackbox

# Multi-platform, custom AXI widths, custom package/design names
init-target --name my-soc \
            --package mysoc --design MySoCTop \
            --bridge uart,fased,blockdev \
            --platform verilator,vcs \
            --feature verilog-blackbox \
            --axi-addr-width 40 --axi-data-width 128 --axi-id-width 6

# Always dry-run first
init-target --name my-soc --bridge uart --dry-run

# Overwrite an existing target
init-target --name my-soc --bridge uart,fased --force
